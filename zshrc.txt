# Silenciar advertencias de Powerlevel10k
typeset -g POWERLEVEL9K_INSTANT_PROMPT=quiet

# Ruta de Oh My Zsh
export ZSH="$HOME/.oh-my-zsh"

# Tema Powerlevel10k
ZSH_THEME="powerlevel10k/powerlevel10k"

autoload -Uz compinit
compinit

# Mejoras en la navegación con TAB
zstyle ':completion:*' menu select
zstyle ':completion:*' matcher-list 'm:{a-z}={A-Za-z}'

# Plugins
# plugins=(git archlinux z sudo)
plugins=(git zsh-autosuggestions zsh-syntax-highlighting)

# Inicialización de Oh My Zsh
source $ZSH/oh-my-zsh.sh

# Powerlevel10k config (no borrar)
[[ ! -f ~/.p10k.zsh ]] || source ~/.p10k.zsh

# Editor por defecto
export EDITOR=nano

# Colores para `ls` y otros comandos
export CLICOLOR=1
export LSCOLORS=GxFxCxDxBxegedabagaced

# Alias útiles
alias update="sudo pacman -Syu"
alias install="sudo pacman -S"
alias remove="sudo pacman -Rns"
alias clean="sudo pacman -Rns \$(pacman -Qdtq)"
alias mirrord="sudo reflector --latest 10 --sort rate --save /etc/pacman.d/mirrorlist"
alias yayupdate="yay -Syu --devel --timeupdate"
alias grep="grep --color=auto"
alias cls="clear"
# Reemplazo moderno de ls con exa
alias ls='exa --color=always --group-directories-first'
alias ll='exa -lah --git --icons'
alias la='exa -a --icons'
alias tree='exa --tree --level=2 --icons'

# Path adicional si usas yay desde /opt o ~/.local/bin
export PATH=$PATH:$HOME/.local/bin:/opt/yay/bin

# Mostrar fastfetch al iniciar la terminal
clear
fastfetch
echo -e "\e[1;36m[ orami - Arch Linux VM ]\e[0m"
